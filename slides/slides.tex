    \section{Modeling}

    \begin{frame}
        \frametitle{Constant Predictor as a Baseline}
        \begin{itemize}
            \item The baseline always predicts majority class, i.e., non-failure of the machine.
            \item High accuracy observed (0.84), but not representative because of class imbalance.
            \item Balanced accuracy of 0.5 was observed, indicating no ability to distinguish failures from non-failures.
            \item PR-AUC of 0.1 observed
        \end{itemize}
    \end{frame}

    \begin{frame}
        \frametitle{Five base models considered}
        \begin{itemize}
                    \item Gaussian Naive Bayes
                    \item Linear discriminant analysis
                    \item K-Nearest Neighbors
                    \item Gradient Boosting
                    \item Logistic regression
        \end{itemize}
    \end{frame}
